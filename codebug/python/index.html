1. Most Frequent Character

Write a function that returns the most frequent character in a string.

import java.util.HashMap;

public class Main {
    public static char mostFrequentChar(String str) {
        HashMap frequencyMap = new HashMap<>();
        for (char c : str.toCharArray()) {
            if (frequencyMap.containsKey(c)) {
                frequencyMap.put(c, frequencyMap.get(c) + 1);
            } else {
                frequencyMap.put(c, 1);
            }
        }

        char mostFrequent = ' ';
        int maxCount = 0;
        for (char c : frequencyMap.keySet()) {
            if (frequencyMap.get(c) > maxCount) {
                maxCount = frequencyMap.get(c);
                mostFrequent = c;
            }
        }
        return mostFrequent;
    }

    public static void main(String[] args) {
        System.out.println(mostFrequentChar("hello world"));
    }
}
